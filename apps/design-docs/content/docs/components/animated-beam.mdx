---
title: "Animated Beam"
description: "Draws an animated glowing beam between two DOM elements. Perfect for integration diagrams, data-flow visualizations, and connection showcases."
---

## Usage

```tsx
import { AnimatedBeam } from "@nebutra/custom-ui/primitives"
```

## Examples

### Basic Connection

```tsx
function BeamDemo() {
  const containerRef = useRef<HTMLDivElement>(null);
  const fromRef = useRef<HTMLDivElement>(null);
  const toRef = useRef<HTMLDivElement>(null);

  return (
    <div ref={containerRef} className="relative flex justify-between p-16">
      <div ref={fromRef} className="rounded-full bg-blue-500 p-4">A</div>
      <div ref={toRef} className="rounded-full bg-purple-500 p-4">B</div>
      <AnimatedBeam
        containerRef={containerRef}
        fromRef={fromRef}
        toRef={toRef}
      />
    </div>
  );
}
```

### Curved Beam

```tsx
<AnimatedBeam
  containerRef={containerRef}
  fromRef={fromRef}
  toRef={toRef}
  curvature={50}
/>
```

### Custom Colors

```tsx
<AnimatedBeam
  containerRef={containerRef}
  fromRef={fromRef}
  toRef={toRef}
  gradientStartColor="#00ff88"
  gradientStopColor="#0066ff"
  pathColor="rgba(0,100,255,0.2)"
/>
```

### Reversed Direction

```tsx
<AnimatedBeam
  containerRef={containerRef}
  fromRef={fromRef}
  toRef={toRef}
  reverse
/>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `containerRef` | `RefObject<HTMLElement>` | — | Ref to the bounding container |
| `fromRef` | `RefObject<HTMLElement>` | — | Start element ref |
| `toRef` | `RefObject<HTMLElement>` | — | End element ref |
| `curvature` | `number` | `0` | Bezier curve height in px (positive = curves down) |
| `reverse` | `boolean` | `false` | Reverse the beam travel direction |
| `duration` | `number` | random 4–7 | Animation duration in seconds |
| `delay` | `number` | `0` | Start delay in seconds |
| `pathColor` | `string` | `"gray"` | Color of the static path line |
| `pathWidth` | `number` | `2` | Width of the path line |
| `pathOpacity` | `number` | `0.2` | Opacity of the static path |
| `gradientStartColor` | `string` | `"#ffaa40"` | Beam gradient start color |
| `gradientStopColor` | `string` | `"#9c40ff"` | Beam gradient stop color |
| `startXOffset` | `number` | `0` | X offset from the start element center |
| `startYOffset` | `number` | `0` | Y offset from the start element center |
| `endXOffset` | `number` | `0` | X offset from the end element center |
| `endYOffset` | `number` | `0` | Y offset from the end element center |
| `className` | `string` | — | Additional CSS classes on the SVG |

## Notes

- The component renders an absolutely-positioned SVG that fills the container.
- Uses `ResizeObserver` to update path on container resize.
- Multiple beams can be layered on the same container.
