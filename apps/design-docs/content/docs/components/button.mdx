## Installation

```tsx
import { Button, ButtonLink } from "@nebutra/custom-ui/primitives"
```

## Variants

```tsx
<Button variant="default">Default</Button>
<Button variant="outline">Outline</Button>
<Button variant="secondary">Secondary</Button>
<Button variant="ghost">Ghost</Button>
<Button variant="destructive">Delete</Button>
<Button variant="warning">Warning</Button>
<Button variant="link">Learn more</Button>
```

## Sizes

```tsx
<Button size="tiny">Tiny</Button>
<Button size="sm">Small</Button>
<Button size="default">Default</Button>
<Button size="lg">Large</Button>
```

## With Icons

```tsx
import { Mail, ArrowRight } from "lucide-react"

<Button prefix={<Mail />}>Login with Email</Button>
<Button suffix={<ArrowRight />}>Continue</Button>
```

## Loading State

```tsx
<Button loading>Saving…</Button>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "destructive" \| "outline" \| "secondary" \| "ghost" \| "link" \| "tertiary" \| "warning"` | `"default"` | Visual style variant |
| `size` | `"tiny" \| "sm" \| "default" \| "lg" \| "icon"` | `"default"` | Size preset |
| `prefix` | `ReactNode` | — | Icon before children |
| `suffix` | `ReactNode` | — | Icon after children |
| `loading` | `boolean` | `false` | Show spinner |
| `asChild` | `boolean` | `false` | Merge props onto immediate child |
| `disabled` | `boolean` | `false` | Disables the button |
