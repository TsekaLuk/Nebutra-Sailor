## Demo

```tsx
import {
  NavigationMenu,
  NavigationMenuList,
  NavigationMenuItem,
  NavigationMenuTrigger,
  NavigationMenuContent,
  NavigationMenuLink,
  navigationMenuTriggerStyle,
} from "@nebutra/ui/primitives"
import Link from "next/link"

export function NavigationMenuDemo() {
  return (
    <NavigationMenu>
      <NavigationMenuList>
        <NavigationMenuItem>
          <NavigationMenuTrigger>Getting Started</NavigationMenuTrigger>
          <NavigationMenuContent>
            <ul className="grid gap-3 p-4 w-96">
              <li>
                <NavigationMenuLink asChild>
                  <Link href="/docs/introduction" className="block p-3 rounded-md hover:bg-accent">
                    <div className="font-medium mb-1">Introduction</div>
                    <p className="text-sm text-muted-foreground">Overview of the design system.</p>
                  </Link>
                </NavigationMenuLink>
              </li>
              <li>
                <NavigationMenuLink asChild>
                  <Link href="/docs/installation" className="block p-3 rounded-md hover:bg-accent">
                    <div className="font-medium mb-1">Installation</div>
                    <p className="text-sm text-muted-foreground">How to install and configure.</p>
                  </Link>
                </NavigationMenuLink>
              </li>
            </ul>
          </NavigationMenuContent>
        </NavigationMenuItem>
        <NavigationMenuItem>
          <NavigationMenuLink asChild>
            <Link href="/docs" className={navigationMenuTriggerStyle()}>
              Documentation
            </Link>
          </NavigationMenuLink>
        </NavigationMenuItem>
      </NavigationMenuList>
    </NavigationMenu>
  )
}
```

## Props

### NavigationMenu (Root)

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Menu orientation |
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Menu content |

### NavigationMenuTrigger

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Trigger label |

### NavigationMenuContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Flyout panel content |

### NavigationMenuLink

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `active` | `boolean` | `false` | Marks as current page |
| `asChild` | `boolean` | `false` | Merge props onto child element |
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Link content |

## Accessibility

- Built on Radix UI Navigation Menu — follows the WAI-ARIA [Disclosure Navigation pattern](https://www.w3.org/WAI/ARIA/apg/patterns/disclosure/examples/disclosure-navigation/).
- `NavigationMenuList` has `role="menubar"`.
- Triggers with flyout content have `aria-expanded` and `aria-haspopup="true"`.
- Active links have `aria-current="page"`.
- Keyboard: `Tab` to navigate items, `Enter`/`Space` to open flyout, `Escape` to close.
