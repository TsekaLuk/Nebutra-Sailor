## Demo

```tsx
import {
  Alert,
  AlertIcon,
  AlertTitle,
  AlertDescription,
  AlertContent,
} from "@nebutra/ui/primitives";
import { CircleCheck, CircleAlert, TriangleAlert, Info } from "lucide-react";

export function AlertDemo() {
  return (
    <div className="space-y-3">
      <Alert variant="success" appearance="light">
        <AlertIcon>
          <CircleCheck />
        </AlertIcon>
        <AlertContent>
          <AlertTitle>Operation successful!</AlertTitle>
          <AlertDescription>Your changes have been saved.</AlertDescription>
        </AlertContent>
      </Alert>

      <Alert variant="destructive" appearance="light">
        <AlertIcon>
          <CircleAlert />
        </AlertIcon>
        <AlertContent>
          <AlertTitle>Something went wrong</AlertTitle>
          <AlertDescription>
            Please try again or contact support.
          </AlertDescription>
        </AlertContent>
      </Alert>

      <Alert variant="warning" appearance="light">
        <AlertIcon>
          <TriangleAlert />
        </AlertIcon>
        <AlertTitle>Storage almost full</AlertTitle>
      </Alert>

      <Alert variant="info" appearance="light">
        <AlertIcon>
          <Info />
        </AlertIcon>
        <AlertTitle>New features available</AlertTitle>
      </Alert>
    </div>
  );
}
```

## Usage

```tsx
import {
  Alert,
  AlertIcon,
  AlertTitle,
  AlertDescription,
  AlertContent,
  AlertToolbar,
} from "@nebutra/ui/primitives";
```

## Props

### Alert

| Prop         | Type                                                                                      | Default       | Description                                     |
| ------------ | ----------------------------------------------------------------------------------------- | ------------- | ----------------------------------------------- |
| `variant`    | `"secondary" \| "primary" \| "destructive" \| "success" \| "info" \| "mono" \| "warning"` | `"secondary"` | Color and semantic intent                       |
| `appearance` | `"solid" \| "outline" \| "light" \| "stroke"`                                             | `"solid"`     | Visual style treatment                          |
| `size`       | `"sm" \| "md" \| "lg"`                                                                    | `"md"`        | Size of the alert                               |
| `icon`       | `"primary" \| "destructive" \| "success" \| "info" \| "warning"`                          | —             | Override icon color (use with `variant="mono"`) |
| `close`      | `boolean`                                                                                 | `false`       | Whether to show the dismiss button              |
| `onClose`    | `() => void`                                                                              | —             | Callback when dismiss button is clicked         |
| `className`  | `string`                                                                                  | —             | Additional CSS classes                          |
| `children`   | `ReactNode`                                                                               | —             | Alert content                                   |

### AlertIcon

| Prop        | Type        | Default | Description                           |
| ----------- | ----------- | ------- | ------------------------------------- |
| `children`  | `ReactNode` | —       | Icon element (e.g. lucide-react icon) |
| `className` | `string`    | —       | Additional CSS classes                |

### AlertTitle

| Prop        | Type        | Default | Description            |
| ----------- | ----------- | ------- | ---------------------- |
| `children`  | `ReactNode` | —       | Title text content     |
| `className` | `string`    | —       | Additional CSS classes |

### AlertDescription

| Prop        | Type        | Default | Description              |
| ----------- | ----------- | ------- | ------------------------ |
| `children`  | `ReactNode` | —       | Descriptive text content |
| `className` | `string`    | —       | Additional CSS classes   |

### AlertContent

| Prop        | Type        | Default | Description                                       |
| ----------- | ----------- | ------- | ------------------------------------------------- |
| `children`  | `ReactNode` | —       | Container for `AlertTitle` and `AlertDescription` |
| `className` | `string`    | —       | Additional CSS classes                            |

### AlertToolbar

| Prop        | Type        | Default | Description             |
| ----------- | ----------- | ------- | ----------------------- |
| `children`  | `ReactNode` | —       | Action buttons or links |
| `className` | `string`    | —       | Additional CSS classes  |

## Examples

### Solid Variants

```tsx
<Alert variant="primary" appearance="solid">
  <AlertTitle>Primary alert — solid appearance</AlertTitle>
</Alert>

<Alert variant="destructive" appearance="solid">
  <AlertTitle>Destructive — solid appearance</AlertTitle>
</Alert>
```

### Dismissible Alert

```tsx
const [visible, setVisible] = useState(true);

{
  visible && (
    <Alert
      variant="info"
      appearance="light"
      close
      onClose={() => setVisible(false)}
    >
      <AlertIcon>
        <Info />
      </AlertIcon>
      <AlertTitle>This alert can be dismissed</AlertTitle>
    </Alert>
  );
}
```

### With Toolbar

```tsx
<Alert variant="warning" appearance="light">
  <AlertIcon>
    <TriangleAlert />
  </AlertIcon>
  <AlertContent>
    <AlertTitle>Action required</AlertTitle>
    <AlertDescription>Please review your billing information.</AlertDescription>
  </AlertContent>
  <AlertToolbar>
    <Button size="sm" variant="outline">
      Review
    </Button>
  </AlertToolbar>
</Alert>
```

### Size Variants

```tsx
<Alert variant="success" size="sm">
  <AlertTitle>Small alert</AlertTitle>
</Alert>

<Alert variant="success" size="md">
  <AlertTitle>Medium alert (default)</AlertTitle>
</Alert>

<Alert variant="success" size="lg">
  <AlertTitle>Large alert</AlertTitle>
</Alert>
```

## Accessibility

- Root element renders with `role="alert"` — screen readers announce content immediately.
- The dismiss button has `aria-label="Dismiss"` for screen reader clarity.
- Icon containers use `data-slot` attributes for targeted styling without affecting semantics.
- Supports keyboard interaction for the close button (`Enter`, `Space`).
