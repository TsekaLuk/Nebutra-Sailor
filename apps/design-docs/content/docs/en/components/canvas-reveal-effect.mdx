## Dependencies

Requires `three` and `@react-three/fiber`:

```bash
pnpm add three @react-three/fiber
```

## Usage

```tsx
import { CanvasRevealEffect } from "@nebutra/ui/primitives"
```

## Examples

### Basic

```tsx
<div className="relative h-[400px] w-full overflow-hidden rounded-xl">
  <CanvasRevealEffect />
</div>
```

### Cyan Dots

```tsx
<CanvasRevealEffect
  animationSpeed={0.5}
  colors={[[0, 255, 255]]}
  dotSize={3}
/>
```

### Multi-color

```tsx
<CanvasRevealEffect
  animationSpeed={0.4}
  colors={[
    [59, 130, 246],   // blue
    [139, 92, 246],   // purple
  ]}
  dotSize={4}
  showGradient
/>
```

### On Hover Card

```tsx
function RevealCard() {
  const [hovered, setHovered] = useState(false);
  return (
    <div
      className="relative h-[300px] w-[400px] rounded-xl border p-8"
      onMouseEnter={() => setHovered(true)}
      onMouseLeave={() => setHovered(false)}
    >
      {hovered && <CanvasRevealEffect animationSpeed={0.3} />}
      <p>Hover to reveal</p>
    </div>
  );
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `animationSpeed` | `number` | `0.4` | Animation speed (0.1 = slow, 1.0 = fast) |
| `opacities` | `number[]` | `[0.3,0.3,0.3,0.5,0.5,0.5,0.8,0.8,0.8,1]` | Dot opacity distribution |
| `colors` | `number[][]` | `[[0,255,255]]` | RGB color arrays for dots |
| `containerClassName` | `string` | — | CSS classes on the container |
| `dotSize` | `number` | — | Dot size in pixels |
| `showGradient` | `boolean` | — | Overlay a radial gradient fade |

## Notes

- Type checking is disabled (`@ts-nocheck`) due to `@react-three/fiber` JSX incompatibilities with React 19. The component works correctly at runtime.
- Wrap in a container with `overflow-hidden` for clean edges.
