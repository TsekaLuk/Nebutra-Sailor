## Usage

```tsx
import {
  Command,
  CommandEmpty,
  CommandGroup,
  CommandInput,
  CommandItem,
  CommandList,
  CommandSeparator,
  CommandShortcut,
  CommandDialog,
} from "@nebutra/ui/primitives"
```

## Props

### Command (Root)

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | ‚Äî | Additional CSS classes |
| `onSelect` | `(value: string) => void` | ‚Äî | Callback when an item is selected |
| `shouldFilter` | `boolean` | `true` | Whether to filter items based on search input |
| `filter` | `(value: string, search: string) => number` | ‚Äî | Custom filter function |
| `children` | `ReactNode` | ‚Äî | Command content |

### CommandInput

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `placeholder` | `string` | ‚Äî | Placeholder text |
| `value` | `string` | ‚Äî | Controlled search value |
| `onValueChange` | `(value: string) => void` | ‚Äî | Callback on input change |
| `className` | `string` | ‚Äî | Additional CSS classes |

### CommandItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | ‚Äî | Value for filtering and selection |
| `disabled` | `boolean` | `false` | Prevents selection |
| `onSelect` | `(value: string) => void` | ‚Äî | Callback when item is selected |
| `className` | `string` | ‚Äî | Additional CSS classes |
| `children` | `ReactNode` | ‚Äî | Item content |

## Examples

### Command Dialog (Modal)

<ComponentPreview className="min-h-[400px] flex justify-center items-start pt-10">
    <CommandDialogDemo />

  ```tsx
import * as React from "react"
import {
  CommandDialog,
  CommandEmpty,
  CommandGroup,
  CommandInput,
  CommandItem,
  CommandList,
} from "@nebutra/ui/primitives"

export function CommandDialogDemo() {
  const [open, setOpen] = React.useState(false)

  React.useEffect(() => {
    const down = (e: KeyboardEvent) => {
      if (e.key === "j" && (e.metaKey || e.ctrlKey)) {
        e.preventDefault()
        setOpen((open) => !open)
      }
    }

    document.addEventListener("keydown", down)
    return () => document.removeEventListener("keydown", down)
  }, [])

  return (
    <>
      <p className="text-sm text-muted-foreground">
        Press{" "}
        <kbd className="pointer-events-none inline-flex h-5 select-none items-center gap-1 rounded border bg-muted px-1.5 font-mono text-[10px] font-medium text-muted-foreground opacity-100">
          <span className="text-xs">‚åò</span>J
        </kbd>
      </p>
      <CommandDialog open={open} onOpenChange={setOpen}>
        <CommandInput placeholder="Type a command or search..." />
        <CommandList>
          <CommandEmpty>No results found.</CommandEmpty>
          <CommandGroup heading="Suggestions">
            <CommandItem onSelect={() => console.log("Calendar")}>
              Calendar
            </CommandItem>
            <CommandItem onSelect={() => console.log("Search Emoji")}>
              Search Emoji
            </CommandItem>
            <CommandItem onSelect={() => console.log("Calculator")}>
              Calculator
            </CommandItem>
          </CommandGroup>
        </CommandList>
      </CommandDialog>
    </>
  )
}
```
</ComponentPreview>

## Accessibility

- Built on cmdk ‚Äî keyboard-first design.
- `CommandInput` has `role="combobox"` with `aria-expanded` and `aria-haspopup="listbox"`.
- `CommandList` has `role="listbox"`.
- Items have `role="option"` with `aria-selected`.
- Keyboard: type to search, `ArrowUp`/`ArrowDown` to navigate, `Enter` to select.
