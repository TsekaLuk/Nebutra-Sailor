<ComponentPreview>

```tsx title="dithering-background-demo.tsx"
"use client";

import { DitheringBackground } from "@nebutra/ui/primitives";
import { useState } from "react";

export function DitheringBackgroundDemo() {
    const [theme, setTheme] = useState<"system" | "light" | "dark">("dark");

    return (
        <div className="relative w-full h-[500px] rounded-xl overflow-hidden border flex flex-col items-center justify-center">
            <DitheringBackground
                themeMode={theme}
                intensity={0.65}
                shape="swirl"
                type="8x8"
                parallax={true}
                parallaxStrength={15}
                className="rounded-xl"
            />

            <div className="z-10 relative flex flex-col items-center bg-background/50 backdrop-blur-md p-6 rounded-2xl border shadow-lg max-w-sm text-center gap-4">
                <h3 className="text-2xl font-bold tracking-tight">Dithering Shader</h3>
                <p className="text-muted-foreground text-sm">
                    A WebGL-powered shader background featuring retro dithering patterns and interactive parallax.
                </p>

                <div className="flex gap-2 mt-2">
                    <button
                        className={`px-3 py-1 text-xs rounded-full border ${theme === 'dark' ? 'bg-primary text-primary-foreground' : 'bg-background hover:bg-muted'}`}
                        onClick={() => setTheme('dark')}
                    >
                        Dark
                    </button>
                    <button
                        className={`px-3 py-1 text-xs rounded-full border ${theme === 'light' ? 'bg-primary text-primary-foreground' : 'bg-background hover:bg-muted'}`}
                        onClick={() => setTheme('light')}
                    >
                        Light
                    </button>
                </div>
            </div>
        </div>
    );
}

```
</ComponentPreview>


## Usage

```tsx
import { DitheringBackground } from "@nebutra/ui/primitives"
```

## Examples

### Basic

```tsx
<div className="relative min-h-screen">
  <DitheringBackground />
  <main className="relative z-10">Content</main>
</div>
```

### Dark Mode Override

```tsx
<div className="relative h-[400px]">
  <DitheringBackground mode="dark" />
</div>
```

### Light Mode Override

```tsx
<div className="relative h-[200px]">
  <DitheringBackground mode="light" />
</div>
```

### System (auto)

```tsx
<DitheringBackground mode="system" />
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `mode` | `"light" \| "dark" \| "system"` | `"system"` | Theme mode — `system` follows `prefers-color-scheme` |
| `className` | `string` | — | Additional CSS classes |

## Notes

- Powered by `@paper-design/shaders-react`'s `Dithering` component.
- Requires WebGL support.
- The component is `absolute inset-0 -z-10` — place it inside a `relative` container.
- Theme colors are predefined in light/dark configs. Custom colors are available via the underlying `DitheringProps`.
