## Demo

<ComponentPreview className="min-h-[400px] flex justify-center items-center">
  <FormDemo />
</ComponentPreview>

```tsx
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { z } from "zod"
import {
  Form,
  FormField,
  FormItem,
  FormLabel,
  FormControl,
  FormDescription,
  FormMessage,
} from "@nebutra/ui/primitives"
import { Input } from "@nebutra/ui/primitives"
import { Button } from "@nebutra/ui/primitives"

const formSchema = z.object({
  username: z.string().min(2, "Username must be at least 2 characters."),
})

export function FormDemo() {
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
    defaultValues: { username: "" },
  })

  function onSubmit(values: z.infer<typeof formSchema>) {
    console.log(values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4 w-80">
        <FormField
          control={form.control}
          name="username"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Username</FormLabel>
              <FormControl>
                <Input placeholder="johndoe" {...field} />
              </FormControl>
              <FormDescription>This is your public display name.</FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  )
}
```

## Usage

```tsx
import {
  Form,
  FormField,
  FormItem,
  FormLabel,
  FormControl,
  FormDescription,
  FormMessage,
} from "@nebutra/ui/primitives"
```

<Info>
  Requires `react-hook-form` and `@hookform/resolvers`. Use `zod` for schema
  validation following the project's input validation standards.
</Info>

## Props

### FormField

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `control` | `Control<T>` | — | **Required.** React Hook Form control object |
| `name` | `string` | — | **Required.** Field name matching the form schema |
| `render` | `({ field }) => ReactNode` | — | **Required.** Render function receiving field props |

### FormItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | `FormLabel`, `FormControl`, `FormDescription`, `FormMessage` |

### FormLabel

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Label text |

### FormControl

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | — | The input element receiving `aria-*` attributes |

### FormMessage

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Override the error message (uses form state by default) |

## Examples

### Multi-field Form

```tsx
const form = useForm<z.infer<typeof schema>>({
  resolver: zodResolver(schema),
  defaultValues: { email: "", password: "" },
})

<Form {...form}>
  <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
    <FormField control={form.control} name="email" render={({ field }) => (
      <FormItem>
        <FormLabel>Email</FormLabel>
        <FormControl><Input type="email" {...field} /></FormControl>
        <FormMessage />
      </FormItem>
    )} />
    <FormField control={form.control} name="password" render={({ field }) => (
      <FormItem>
        <FormLabel>Password</FormLabel>
        <FormControl><Input type="password" {...field} /></FormControl>
        <FormMessage />
      </FormItem>
    )} />
    <Button type="submit">Sign in</Button>
  </form>
</Form>
```

## Accessibility

- `FormLabel` is linked to its control via generated `id`/`htmlFor`.
- `FormControl` injects `aria-describedby` (pointing to description/message) and `aria-invalid` on error.
- `FormMessage` has `aria-live="polite"` for dynamic error announcement.
- Error messages are announced to screen readers without page refresh.
