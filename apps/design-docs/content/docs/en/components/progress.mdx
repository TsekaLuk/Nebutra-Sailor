## Usage

```tsx
import { Progress } from "@nebutra/ui/primitives"
```

## Props

Extends `@radix-ui/react-progress` `Root` props.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number \| null` | — | Progress value from `0` to `max`. `null` for indeterminate |
| `max` | `number` | `100` | Maximum value |
| `getValueLabel` | `(value: number, max: number) => string` | — | Custom label for screen readers |
| `className` | `string` | — | Additional CSS classes |

## Examples

### Animated Progress

<ComponentPreview name="progress-1-demo" className="max-w-[400px]">
  <Progress1Demo />

  ```tsx
import * as React from "react"
import { Progress } from "@nebutra/ui/primitives"

export function ProgressAnimatedDemo() {
  const [progress, setProgress] = React.useState(13)

  React.useEffect(() => {
    const timer = setTimeout(() => setProgress(65), 500)
    return () => clearTimeout(timer)
  }, [])

  return (
    <div className="w-full">
      <Progress value={progress} className="w-full" />
    </div>
  )
}
```

### Circular Progress

<ComponentPreview name="animated-circular-progress-bar-demo" className="max-w-[400px]">
  <AnimatedCircularProgressBarDemo />

  ```tsx
import { useEffect, useState } from "react"
import { AnimatedCircularProgressBar } from "@nebutra/ui/primitives"

export function AnimatedCircularProgressBarDemo() {
  const [value, setValue] = useState(0)

  useEffect(() => {
    const handleIncrement = (prev: number) => {
      if (prev === 100) return 0
      return prev + 10
    }
    setValue(handleIncrement)
    const interval = setInterval(() => setValue(handleIncrement), 2000)
    return () => clearInterval(interval)
  }, [])

  return (
    <AnimatedCircularProgressBar
      max={100}
      min={0}
      value={value}
      gaugePrimaryColor="hsl(var(--primary))"
    />
  )
}
```
</ComponentPreview>

## Accessibility

- Built on Radix UI Progress — has `role="progressbar"`.
- Sets `aria-valuemin="0"`, `aria-valuemax` (default 100), and `aria-valuenow` based on `value`.
- `aria-label` or `aria-labelledby` should be added for context.
- Indeterminate state (`value={null}`) omits `aria-valuenow`.
