## Demo

```tsx
import { RadioGroupCard, RadioGroupCardItem } from "@nebutra/ui/primitives"
import { CreditCard, Landmark, Smartphone } from "lucide-react"

export function RadioGroupCardDemo() {
  return (
    <RadioGroupCard defaultValue="card" className="grid grid-cols-3 gap-3">
      <RadioGroupCardItem value="card" id="pay-card">
        <CreditCard className="mb-2 h-6 w-6" />
        <div className="font-medium">Card</div>
        <div className="text-xs text-muted-foreground">Visa, Mastercard</div>
      </RadioGroupCardItem>
      <RadioGroupCardItem value="bank" id="pay-bank">
        <Landmark className="mb-2 h-6 w-6" />
        <div className="font-medium">Bank</div>
        <div className="text-xs text-muted-foreground">Direct transfer</div>
      </RadioGroupCardItem>
      <RadioGroupCardItem value="mobile" id="pay-mobile">
        <Smartphone className="mb-2 h-6 w-6" />
        <div className="font-medium">Mobile</div>
        <div className="text-xs text-muted-foreground">Apple Pay, Google Pay</div>
      </RadioGroupCardItem>
    </RadioGroupCard>
  )
}
```

## Usage

```tsx
import { RadioGroupCard, RadioGroupCardItem } from "@nebutra/ui/primitives"
```

## Props

### RadioGroupCard

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | Controlled selected value |
| `defaultValue` | `string` | — | Uncontrolled default selected value |
| `onValueChange` | `(value: string) => void` | — | Callback when selection changes |
| `disabled` | `boolean` | `false` | Disables all items |
| `className` | `string` | — | Additional CSS classes (use `grid` utilities for layout) |
| `children` | `ReactNode` | — | `RadioGroupCardItem` elements |

### RadioGroupCardItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | **Required.** Value for this option |
| `id` | `string` | — | For accessibility linking |
| `disabled` | `boolean` | `false` | Disables this specific item |
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Card content (icon + label + description) |

## Accessibility

- Selected item has a visible checked indicator and border highlight.
- Each card item has `role="radio"` with `aria-checked`.
- Keyboard: `ArrowLeft`/`ArrowRight` or `ArrowUp`/`ArrowDown` to navigate between cards.
- `Space` to select the focused card.
