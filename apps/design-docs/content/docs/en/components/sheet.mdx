```tsx
  Sheet,
  SheetTrigger,

  SheetHeader,
  SheetTitle,
  SheetDescription,
  SheetFooter,
  SheetClose,
  SheetPortal,
  SheetOverlay,
} from "@nebutra/ui/primitives"
```

## Props

### Sheet (Root)

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | — | Controlled open state |
| `defaultOpen` | `boolean` | `false` | Uncontrolled default open state |
| `onOpenChange` | `(open: boolean) => void` | — | Callback when open state changes |

### SheetContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `side` | `"top" \| "right" \| "bottom" \| "left"` | `"right"` | Edge the sheet slides in from |
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Sheet content |

## Accessibility

- Built on Radix UI Dialog — `SheetContent` has `role="dialog"` and `aria-modal="true"`.
- `SheetTitle` is linked via `aria-labelledby`.
- Focus is trapped inside the sheet while open.
- `Escape` closes the sheet and returns focus to the trigger.
