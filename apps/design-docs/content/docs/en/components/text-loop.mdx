---
title: "Text Loop"
description: "Cycles through an array of text/content items with smooth Framer Motion slide animations. Ideal for rotating taglines, testimonials, or hero subtitles."
---

## Usage

```tsx
import { TextLoop } from "@nebutra/ui/primitives"
```

## Examples

### Basic Rotating Words

```tsx
<TextLoop>
  {["Designers", "Developers", "Founders"].map(t => (
    <span key={t}>{t}</span>
  ))}
</TextLoop>
```

### Inline With Static Text

```tsx
<p className="text-2xl">
  Built for{" "}
  <TextLoop interval={1.5}>
    {["speed", "scale", "simplicity"].map(t => (
      <span key={t} className="font-bold text-primary">{t}</span>
    ))}
  </TextLoop>
</p>
```

### Custom Transition

```tsx
<TextLoop
  interval={3}
  transition={{ duration: 0.5, ease: "easeInOut" }}
>
  {["Hello", "Bonjour", "Hola", "Ciao"].map(t => (
    <span key={t}>{t}</span>
  ))}
</TextLoop>
```

### With Index Callback

```tsx
<TextLoop onIndexChange={(i) => console.warn("index:", i)}>
  {items.map(item => <span key={item.id}>{item.label}</span>)}
</TextLoop>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode[]` | — | Array of items to cycle through |
| `interval` | `number` | `2` | Seconds between each transition |
| `transition` | `Transition` | `{ duration: 0.3 }` | Framer Motion transition config |
| `variants` | `Variants` | slide up/down | Custom Framer Motion animation variants |
| `onIndexChange` | `(index: number) => void` | — | Callback when active index changes |
| `className` | `string` | — | Additional CSS classes |

## Default Animation

Items slide up on enter and slide up on exit (vertical scroll wheel feel):

```
initial: { y: 20, opacity: 0 }
animate: { y: 0, opacity: 1 }
exit:    { y: -20, opacity: 0 }
```
