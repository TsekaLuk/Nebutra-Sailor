## Usage

```tsx
import { ThemeSwitcher } from "@nebutra/ui";

// Uncontrolled
<ThemeSwitcher defaultValue="system" onChange={(theme) => console.log(theme)} />

// Controlled
const [theme, setTheme] = useState("system");
<ThemeSwitcher value={theme} onChange={setTheme} />
```

## Examples

### With next-themes

The primary integration pattern — connects directly to `next-themes`:

```tsx
import { useTheme } from "next-themes";
import { ThemeSwitcher } from "@nebutra/ui";

function AppThemeSwitcher() {
  const { theme, setTheme } = useTheme();

  return (
    <ThemeSwitcher
      value={theme as "light" | "dark" | "system"}
      onChange={setTheme}
    />
  );
}
```

### Standalone Controlled

```tsx
const [theme, setTheme] = useState<"light" | "dark" | "system">("system");

<ThemeSwitcher value={theme} onChange={setTheme} />
```

### Uncontrolled with Default

```tsx
<ThemeSwitcher defaultValue="dark" />
```

### Custom Icons

Replace any of the default icons (Monitor, Sun, Moon) with custom ones:

```tsx
import { Laptop, Sparkles, CloudMoon } from "lucide-react";

<ThemeSwitcher
  icons={{
    system: Laptop,
    light: Sparkles,
    dark: CloudMoon,
  }}
/>
```

### Custom Accessibility Labels

```tsx
<ThemeSwitcher
  labels={{
    system: "Match device preference",
    light: "Switch to light mode",
    dark: "Switch to dark mode",
  }}
/>
```

### In a Navbar

```tsx
<nav className="flex items-center justify-between px-6 py-3">
  <Logo />
  <div className="flex items-center gap-4">
    <Links />
    <ThemeSwitcher value={theme} onChange={setTheme} />
  </div>
</nav>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `"light" \| "dark" \| "system"` | — | Controlled theme value |
| `defaultValue` | `"light" \| "dark" \| "system"` | `"system"` | Initial value for uncontrolled mode |
| `onChange` | `(theme: ThemeSwitcherValue) => void` | — | Callback fired when user selects a theme |
| `icons` | `{ system?: ElementType; light?: ElementType; dark?: ElementType }` | Monitor, Sun, Moon | Custom icon components for each mode |
| `labels` | `{ system?: string; light?: string; dark?: string }` | — | Accessibility labels for each button |
| `className` | `string` | — | Additional Tailwind classes for the container |

## Notes

- **Hydration-safe**: The component renders a skeleton placeholder server-side and during first render to prevent SSR hydration mismatches. The actual theme buttons only appear after mount.
- Use `value` + `onChange` for controlled mode (integrating with next-themes or your own state).
- Use `defaultValue` alone for uncontrolled mode where ThemeSwitcher manages its own state.
- Icons receive `className` for color and size (`h-3.5 w-3.5`) — custom icon components must accept these props.
