## When to Use

Use a table pattern when displaying structured data where:
- Users need to compare values across rows and columns
- Data has a defined schema (columns are consistent across all rows)
- Users may need to sort, filter, or act on individual rows

Use a table over a card list when:
- There are 4 or more attributes per item that benefit from columnar alignment
- Sort and filter by column are required

Do not use a table when:
- Items are heterogeneous (different attributes per item) — use a list or cards
- There are fewer than 3 columns — use a definition list or `Card` grid

## Structure

```
Section (title + description + table-level actions)
  ├── FilterBar (search + filter controls, optional)
  ├── DataTable | Table
  │    ├── Table.Head
  │    │    └── Table.Row → Table.Header (per column, sortable)
  │    └── Table.Body
  │         └── Table.Row (per data row)
  │              ├── Table.Cell (data values)
  │              └── Table.Cell → ActionMenu (row actions)
  ├── EmptyState (when data is empty)
  └── Pagination (when data exceeds page size)
```

## Composition

### DataTable (recommended)

`DataTable` from `@primer/react/experimental` is the primary table component. It handles column definition, sorting, and accessibility out of the box.

```tsx
import { DataTable, Table } from "@nebutra/ui";

const columns = [
  {
    header: "Name",
    field: "name",
    sortBy: "alphanumeric",
    renderCell: (row) => <strong>{row.name}</strong>,
  },
  {
    header: "Created",
    field: "createdAt",
    sortBy: "datetime",
    renderCell: (row) => <RelativeTime date={new Date(row.createdAt)} />,
  },
  {
    header: "Status",
    field: "status",
    renderCell: (row) => <StateLabel status={row.status}>{row.status}</StateLabel>,
  },
  {
    header: "",
    id: "actions",
    renderCell: (row) => <RowActionMenu row={row} />,
  },
];

<Table.Container>
  <DataTable
    aria-labelledby="table-heading"
    data={rows}
    columns={columns}
  />
</Table.Container>
```

### Sorting

Pass `sortBy` on the column definition and optionally provide `initialSortColumn` and `initialSortDirection` to `DataTable`.

```tsx
<DataTable
  data={rows}
  columns={columns}
  initialSortColumn="createdAt"
  initialSortDirection="DESC"
/>
```

For server-side sorting, handle the `onSortChange` callback and refetch data with the new sort params.

### Row actions

Row-level actions live in the last column, right-aligned. Use `ActionMenu` for 2 or more actions. For a single primary action, use a `Button variant="invisible"`.

```tsx
function RowActionMenu({ row }: { row: ApiKey }) {
  return (
    <ActionMenu>
      <ActionMenu.Button size="small" variant="invisible">Actions</ActionMenu.Button>
      <ActionMenu.Overlay>
        <ActionList>
          <ActionList.Item onSelect={() => onEdit(row)}>Edit</ActionList.Item>
          <ActionList.Divider />
          <ActionList.Item variant="danger" onSelect={() => onRevoke(row)}>
            Revoke
          </ActionList.Item>
        </ActionList>
      </ActionMenu.Overlay>
    </ActionMenu>
  );
}
```

### Filtering and search

Place a `FilterBar` fragment above the table when users need to filter or search the data set. The filter bar connects to table data via shared state.

```tsx
import { FilterBar } from "@nebutra/ui";

const [search, setSearch] = useState("");
const [statusFilter, setStatusFilter] = useState<string[]>([]);

const filteredRows = rows.filter((row) => {
  const matchesSearch = row.name.toLowerCase().includes(search.toLowerCase());
  const matchesStatus = !statusFilter.length || statusFilter.includes(row.status);
  return matchesSearch && matchesStatus;
});

<FilterBar
  search={search}
  onSearchChange={setSearch}
  filters={[
    {
      label: "Status",
      options: ["active", "revoked"],
      value: statusFilter,
      onChange: setStatusFilter,
    },
  ]}
/>
<Table.Container>
  <DataTable data={filteredRows} columns={columns} />
</Table.Container>
```

### Pagination

Render `Pagination` directly below the `Table.Container`. Show item count alongside (`Showing 1–25 of 142`).

```tsx
<Table.Container>
  <DataTable data={pageData} columns={columns} />
  <Table.Divider />
  <Box padding={3} display="flex" justifyContent="space-between" alignItems="center">
    <Text fontSize={0} color="fg.muted">
      Showing {startItem}–{endItem} of {totalCount}
    </Text>
    <Pagination
      pageCount={totalPages}
      currentPage={page}
      onPageChange={(p) => setPage(p)}
    />
  </Box>
</Table.Container>
```

### Empty and loading states

Render `LoadingState` and `EmptyState` inside the `Table.Container` to preserve column widths and table chrome while data is absent.

```tsx
<Table.Container>
  {isLoading ? (
    <LoadingState size="medium" />
  ) : rows.length === 0 ? (
    <EmptyState
      title="No API keys"
      description="Create a key to start making requests."
      actions={<Button variant="primary" onClick={onCreate}>Create key</Button>}
    />
  ) : (
    <DataTable data={rows} columns={columns} />
  )}
</Table.Container>
```

## Examples

### API keys table

```tsx
function ApiKeysTable() {
  const { data, isLoading, error, refetch } = useApiKeys();
  const [page, setPage] = useState(1);
  const PAGE_SIZE = 25;

  const columns = [
    { header: "Name", field: "name", sortBy: "alphanumeric" as const },
    { header: "Created", field: "createdAt", renderCell: (row) => <RelativeTime date={new Date(row.createdAt)} /> },
    { header: "Last used", field: "lastUsedAt", renderCell: (row) => row.lastUsedAt ? <RelativeTime date={new Date(row.lastUsedAt)} /> : <Text color="fg.muted">Never</Text> },
    { header: "", id: "actions", renderCell: (row) => <RowActionMenu row={row} onRevoke={revoke} /> },
  ];

  if (error) return <ErrorState error={error} onRetry={refetch} />;

  return (
    <Section title="API Keys" actions={<Button variant="primary" onClick={onCreateKey}>New key</Button>}>
      <Table.Container>
        {isLoading ? (
          <LoadingState size="medium" />
        ) : !data?.length ? (
          <EmptyState title="No keys yet" description="Create your first API key." />
        ) : (
          <DataTable
            aria-label="API keys"
            data={data.slice((page - 1) * PAGE_SIZE, page * PAGE_SIZE)}
            columns={columns}
            initialSortColumn="createdAt"
            initialSortDirection="DESC"
          />
        )}
      </Table.Container>
      {data && data.length > PAGE_SIZE && (
        <Pagination
          pageCount={Math.ceil(data.length / PAGE_SIZE)}
          currentPage={page}
          onPageChange={setPage}
        />
      )}
    </Section>
  );
}
```
