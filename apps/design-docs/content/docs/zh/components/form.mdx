## 演示 (Demo)

<ComponentPreview className="min-h-[400px] flex justify-center items-center">
  <FormDemo />
</ComponentPreview>

```tsx
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { z } from "zod"
import {
  Form,
  FormField,
  FormItem,
  FormLabel,
  FormControl,
  FormDescription,
  FormMessage,
} from "@nebutra/ui/primitives"
import { Input } from "@nebutra/ui/primitives"
import { Button } from "@nebutra/ui/primitives"

const formSchema = z.object({
  username: z.string().min(2, "用户名必须至少包含 2 个字符。(Username must be at least 2 characters.)"),
})

export function FormDemo() {
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
    defaultValues: { username: "" },
  })

  function onSubmit(values: z.infer<typeof formSchema>) {
    console.log(values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4 w-80">
        <FormField
          control={form.control}
          name="username"
          render={({ field }) => (
            <FormItem>
              <FormLabel>用户名 (Username)</FormLabel>
              <FormControl>
                <Input placeholder="johndoe" {...field} />
              </FormControl>
              <FormDescription>这是您的公开显示名称。 (This is your public display name.)</FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">提交 (Submit)</Button>
      </form>
    </Form>
  )
}
```

## 使用方法 (Usage)

```tsx
import {
  Form,
  FormField,
  FormItem,
  FormLabel,
  FormControl,
  FormDescription,
  FormMessage,
} from "@nebutra/ui/primitives"
```

<Info>
  需要依赖 `react-hook-form` 和 `@hookform/resolvers`。为了遵循项目的输入验证标准，请使用 `zod` 进行 Schema 验证。
</Info>

## 属性 (Props)

### FormField

| 属性 | 类型 | 默认值 | 描述 |
|------|------|---------|-------------|
| `control` | `Control<T>` | — | **必填。** React Hook Form 控制对象 |
| `name` | `string` | — | **必填。** 与表单 Schema 匹配的字段名称 |
| `render` | `({ field }) => ReactNode` | — | **必填。** 接收 field 属性的渲染函数 |

### FormItem

| 属性 | 类型 | 默认值 | 描述 |
|------|------|---------|-------------|
| `className` | `string` | — | 额外的 CSS 类名 |
| `children` | `ReactNode` | — | 包含 `FormLabel`、`FormControl`、`FormDescription`、`FormMessage` |

### FormLabel

| 属性 | 类型 | 默认值 | 描述 |
|------|------|---------|-------------|
| `className` | `string` | — | 额外的 CSS 类名 |
| `children` | `ReactNode` | — | 标签文本 |

### FormControl

| 属性 | 类型 | 默认值 | 描述 |
|------|------|---------|-------------|
| `children` | `ReactNode` | — | 接收 `aria-*` 属性的输入框元素 |

### FormMessage

| 属性 | 类型 | 默认值 | 描述 |
|------|------|---------|-------------|
| `className` | `string` | — | 额外的 CSS 类名 |
| `children` | `ReactNode` | — | 覆盖错误消息（默认使用表单状态中的错误） |

## 示例 (Examples)

### 多字段表单 (Multi-field Form)

```tsx
const form = useForm<z.infer<typeof schema>>({
  resolver: zodResolver(schema),
  defaultValues: { email: "", password: "" },
})

<Form {...form}>
  <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
    <FormField control={form.control} name="email" render={({ field }) => (
      <FormItem>
        <FormLabel>电子邮箱 (Email)</FormLabel>
        <FormControl><Input type="email" {...field} /></FormControl>
        <FormMessage />
      </FormItem>
    )} />
    <FormField control={form.control} name="password" render={({ field }) => (
      <FormItem>
        <FormLabel>密码 (Password)</FormLabel>
        <FormControl><Input type="password" {...field} /></FormControl>
        <FormMessage />
      </FormItem>
    )} />
    <Button type="submit">登录 (Sign in)</Button>
  </form>
</Form>
```

## 可访问性 (Accessibility)

- `FormLabel` 通过生成的 `id`/`htmlFor` 链接到其控件。
- `FormControl` 会注入 `aria-describedby` （指向描述/消息），并在发生错误时注入 `aria-invalid`。
- `FormMessage` 带有 `aria-live="polite"` 属性，用于动态朗读错误信息。
- 错误消息无需刷新页面即可由屏幕阅读器播报。
