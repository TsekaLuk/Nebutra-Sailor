## Usage

```tsx
import { Admonition } from "@nebutra/ui";

export function GettingStarted() {
  return (
    <Admonition type="info" title="Tip">
      Use keyboard shortcut <kbd>Cmd+K</kbd> to open the command palette.
    </Admonition>
  );
}
```

## Props

| Prop | Type | Default | Description |
|---|---|---|---|
| `type` | `"info" \| "warning" \| "error" \| "success"` | `"info"` | Controls the color scheme and default icon |
| `title` | `string` | — | Bold heading rendered above the body |
| `icon` | `React.ReactNode` | Auto | Override the default type icon |
| `dismissible` | `boolean` | `false` | Renders a close button; fires `onDismiss` when clicked |
| `onDismiss` | `() => void` | — | Callback when the dismiss button is activated |
| `className` | `string` | — | Additional Tailwind classes on the root element |
| `children` | `React.ReactNode` | — | Body content (supports JSX) |

## Examples

### Without a title

```tsx
<Admonition type="warning">
  Deleting a workspace is permanent and cannot be undone.
</Admonition>
```

### Dismissible

```tsx
const [visible, setVisible] = useState(true);

{visible && (
  <Admonition
    type="info"
    title="New feature available"
    dismissible
    onDismiss={() => setVisible(false)}
  >
    Try the new AI-assisted filter suggestions in the query builder.
  </Admonition>
)}
```

### Custom icon

```tsx
import { ShieldAlert } from "lucide-react";

<Admonition
  type="warning"
  title="Security notice"
  icon={<ShieldAlert className="w-4 h-4" />}
>
  Two-factor authentication is not enabled on your account.
</Admonition>
```

## Accessibility

- Root element renders as `role="note"` for informational variants and `role="alert"` for `error` type.
- `dismissible` close button has `aria-label="Dismiss"` and is reachable via keyboard (`Tab`, `Enter`, `Space`).
- Icons are `aria-hidden="true"` — meaning is conveyed by color, icon, and text together.
- Do not rely solely on color to convey the admonition type; the `title` prop should make the intent explicit.
