---
title: "TextConfirmDialog"
description: "Type-to-confirm dialog that requires the user to type a specific string before a destructive action is enabled."
---

## Demo

```tsx
<TextConfirmDialog
  open={isOpen}
  onOpenChange={setIsOpen}
  title="Delete workspace"
  description={
    <>
      This will permanently delete the <strong>Acme Corp</strong> workspace
      and all of its data. This action cannot be undone.
    </>
  }
  confirmText="Acme Corp"
  confirmLabel="I understand, delete this workspace"
  onConfirm={handleDelete}
/>
```

## Usage

```tsx
import { TextConfirmDialog } from "@nebutra/ui";
import { useState } from "react";

export function DangerZone({ workspace }: { workspace: Workspace }) {
  const [open, setOpen] = useState(false);
  const [deleting, setDeleting] = useState(false);

  async function handleDelete() {
    setDeleting(true);
    try {
      await deleteWorkspace(workspace.id);
      router.push("/");
    } finally {
      setDeleting(false);
      setOpen(false);
    }
  }

  return (
    <>
      <Button variant="danger" onClick={() => setOpen(true)}>
        Delete workspace
      </Button>

      <TextConfirmDialog
        open={open}
        onOpenChange={setOpen}
        title="Delete workspace"
        description={`Type "${workspace.name}" to confirm deletion.`}
        confirmText={workspace.name}
        confirmLabel="Delete workspace"
        onConfirm={handleDelete}
        isLoading={deleting}
      />
    </>
  );
}
```

## Props

| Prop | Type | Default | Description |
|---|---|---|---|
| `open` | `boolean` | — | Controls dialog open state |
| `onOpenChange` | `(open: boolean) => void` | — | Called when the dialog requests a state change |
| `title` | `string` | — | **Required.** Dialog heading |
| `description` | `string \| React.ReactNode` | — | Explanatory content displayed above the input |
| `confirmText` | `string` | — | **Required.** The exact string the user must type to enable the confirm button |
| `confirmLabel` | `string` | `"Confirm"` | Label for the confirm button |
| `cancelLabel` | `string` | `"Cancel"` | Label for the cancel button |
| `inputLabel` | `string` | `"Type {confirmText} to confirm"` | Visible label above the text input |
| `onConfirm` | `() => void \| Promise<void>` | — | Called when the user clicks the confirm button with correct input |
| `onCancel` | `() => void` | — | Called when the user cancels |
| `isLoading` | `boolean` | `false` | Disables all controls and shows a spinner on the confirm button |
| `className` | `string` | — | Additional classes on the dialog panel |

## Accessibility

- Renders as `<dialog>` with `role="alertdialog"`.
- `aria-labelledby` references the title; `aria-describedby` references the description.
- Focus is placed on the text input when the dialog opens.
- The confirm button has `aria-disabled="true"` (not `disabled`) until the text matches, ensuring it remains reachable by screen readers with a relevant tooltip.
- Input has `autocomplete="off"`, `autocorrect="off"`, and `spellcheck="false"` to prevent browser auto-fill.
- Pressing `Escape` closes the dialog, resets the input, and calls `onCancel`.
- The match check is case-sensitive by default to prevent accidental confirmation.
