## When to Use

Use the form pattern whenever you collect structured data from a user and need to validate or submit it. This pattern covers:

- Settings panels (account, billing, notification preferences)
- Create / edit flows (new resource, inline edit)
- Authentication forms (sign-in, sign-up)
- Confirmation dialogs that require typed input

Do not apply this pattern to single-field quick-edits. For a single toggle or a single select that saves on change, use the bare atom component (`Switch`, `Select`) without form scaffolding.

## Structure

```
<Form> (react-hook-form + zod)
  └── FormField (one per field)
       └── FormItem
            ├── FormLabel
            ├── FormControl
            │    └── Input | Textarea | Select | Checkbox | RadioGroup
            ├── FormDescription (hint text)
            └── FormMessage (validation error)
  └── Button type="submit"
```

## Composition

### Basic Field

Every field follows the same `FormField → FormItem → FormControl` structure. `FormMessage` renders the validation error automatically from the schema.

```tsx
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";
import {
  Form,
  FormField,
  FormItem,
  FormLabel,
  FormControl,
  FormDescription,
  FormMessage,
  Input,
  Button,
} from "@nebutra/ui";

const schema = z.object({
  name: z.string().min(1, "Name is required"),
});

export function CreateKeyForm({ onSubmit }: { onSubmit: (v: z.infer<typeof schema>) => void }) {
  const form = useForm({ resolver: zodResolver(schema) });

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        <FormField
          control={form.control}
          name="name"
          render={({ field }) => (
            <FormItem>
              <FormLabel>API key name</FormLabel>
              <FormControl>
                <Input placeholder="e.g. production-server" {...field} />
              </FormControl>
              <FormDescription>
                Used to identify this key in audit logs.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Create key</Button>
      </form>
    </Form>
  );
}
```

### Validation

Schema is the single source of validation truth. `FormMessage` surfaces errors automatically — no manual error state needed.

```tsx
const schema = z.object({
  email: z.string().email("Enter a valid email address"),
  age: z.number().min(18, "Must be 18 or older"),
});
```

Validate on `onBlur` for new forms. Switch to `onChange` after first submission for real-time correction feedback.

```tsx
const form = useForm({
  resolver: zodResolver(schema),
  mode: "onBlur",
});
```

### Select Field

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@nebutra/ui";

<FormField
  control={form.control}
  name="permission"
  render={({ field }) => (
    <FormItem>
      <FormLabel>Permission</FormLabel>
      <Select onValueChange={field.onChange} defaultValue={field.value}>
        <FormControl>
          <SelectTrigger>
            <SelectValue placeholder="Select a permission" />
          </SelectTrigger>
        </FormControl>
        <SelectContent>
          <SelectItem value="read">Read only</SelectItem>
          <SelectItem value="write">Read + write</SelectItem>
        </SelectContent>
      </Select>
      <FormMessage />
    </FormItem>
  )}
/>
```

### Checkbox Field

```tsx
import { Checkbox } from "@nebutra/ui";

<FormField
  control={form.control}
  name="acceptTerms"
  render={({ field }) => (
    <FormItem className="flex items-start gap-3">
      <FormControl>
        <Checkbox checked={field.value} onCheckedChange={field.onChange} />
      </FormControl>
      <div>
        <FormLabel>Accept terms and conditions</FormLabel>
        <FormMessage />
      </div>
    </FormItem>
  )}
/>
```

### Submit and Cancel

Place submit/cancel at the end of the form. Destructive actions go in a separate section with an explicit warning above the button.

```tsx
<div className="flex gap-2 pt-4">
  <Button type="submit" disabled={form.formState.isSubmitting}>
    {form.formState.isSubmitting ? "Saving…" : "Save changes"}
  </Button>
  <Button type="button" variant="ghost" onClick={onCancel}>
    Cancel
  </Button>
</div>
```

### Required Fields

Annotate the label directly — do not rely on asterisk-only conventions without an explanatory legend.

```tsx
<FormItem>
  <FormLabel>
    Organization name <span aria-hidden="true" className="text-destructive">*</span>
  </FormLabel>
  <FormControl>
    <Input required {...field} />
  </FormControl>
  <FormMessage />
</FormItem>
```

### Complex Input Types

| Use case | Component |
|---|---|
| Long text | `Textarea` |
| Single choice from a list | `Select` or `RadioGroup` |
| Binary on/off | `Checkbox` or `Switch` |
| Multi-selection | `multi-select` fragment |
| Date | `DatePicker` |
| OTP / pin | `InputOtp` |

## Examples

### Create API Key Form

```tsx
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";
import {
  Form, FormField, FormItem, FormLabel, FormControl, FormMessage,
  Input, Button,
  Select, SelectTrigger, SelectValue, SelectContent, SelectItem,
} from "@nebutra/ui";

const schema = z.object({
  name: z.string().min(1, "Key name is required"),
  permission: z.enum(["read", "write"]),
});

type FormValues = z.infer<typeof schema>;

export function CreateKeyForm({
  onSubmit,
  onCancel,
}: {
  onSubmit: (v: FormValues) => void;
  onCancel: () => void;
}) {
  const form = useForm<FormValues>({
    resolver: zodResolver(schema),
    defaultValues: { name: "", permission: "read" },
  });

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        <FormField
          control={form.control}
          name="name"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Name</FormLabel>
              <FormControl>
                <Input placeholder="e.g. production-server" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />

        <FormField
          control={form.control}
          name="permission"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Permission</FormLabel>
              <Select onValueChange={field.onChange} defaultValue={field.value}>
                <FormControl>
                  <SelectTrigger>
                    <SelectValue />
                  </SelectTrigger>
                </FormControl>
                <SelectContent>
                  <SelectItem value="read">Read only</SelectItem>
                  <SelectItem value="write">Read + write</SelectItem>
                </SelectContent>
              </Select>
              <FormMessage />
            </FormItem>
          )}
        />

        <div className="flex gap-2 pt-2">
          <Button type="submit">Create key</Button>
          <Button type="button" variant="ghost" onClick={onCancel}>
            Cancel
          </Button>
        </div>
      </form>
    </Form>
  );
}
```
