## Demo

```tsx
import {
  HoverCard,
  HoverCardTrigger,
  HoverCardContent,
} from "@nebutra/custom-ui/primitives"
import { Avatar, AvatarFallback, AvatarImage } from "@nebutra/custom-ui/primitives"

export function HoverCardDemo() {
  return (
    <HoverCard>
      <HoverCardTrigger asChild>
        <a href="#" className="text-sm font-medium underline underline-offset-4">
          @johndoe
        </a>
      </HoverCardTrigger>
      <HoverCardContent className="w-80">
        <div className="flex gap-4">
          <Avatar>
            <AvatarImage src="https://github.com/johndoe.png" />
            <AvatarFallback>JD</AvatarFallback>
          </Avatar>
          <div className="space-y-1">
            <h4 className="text-sm font-semibold">@johndoe</h4>
            <p className="text-sm text-muted-foreground">
              Full-stack engineer. Building great products.
            </p>
            <p className="text-xs text-muted-foreground">Joined January 2022</p>
          </div>
        </div>
      </HoverCardContent>
    </HoverCard>
  )
}
```

## Usage

```tsx
import {
  HoverCard,
  HoverCardTrigger,
  HoverCardContent,
} from "@nebutra/custom-ui/primitives"
```

## Props

### HoverCard (Root)

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | — | Controlled open state |
| `onOpenChange` | `(open: boolean) => void` | — | Callback when open state changes |
| `openDelay` | `number` | `700` | Delay in ms before opening |
| `closeDelay` | `number` | `300` | Delay in ms before closing |

### HoverCardTrigger

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Merge props onto the immediate child element |
| `children` | `ReactNode` | — | Element that triggers the hover card |

### HoverCardContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `side` | `"top" \| "right" \| "bottom" \| "left"` | `"bottom"` | Preferred side |
| `align` | `"start" \| "center" \| "end"` | `"center"` | Alignment |
| `sideOffset` | `number` | `4` | Distance from trigger |
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Card content |

## Examples

### Link Preview Card

```tsx
<HoverCard>
  <HoverCardTrigger asChild>
    <Button variant="link">Visit documentation</Button>
  </HoverCardTrigger>
  <HoverCardContent side="top" className="w-72">
    <div className="space-y-2">
      <h4 className="font-semibold">Nebutra Docs</h4>
      <p className="text-sm text-muted-foreground">
        Comprehensive guides and API references for building with Nebutra.
      </p>
    </div>
  </HoverCardContent>
</HoverCard>
```

## Accessibility

- Built on Radix UI HoverCard.
- HoverCard content is not shown to screen readers who navigate by tab — it is purely visual.
- For keyboard users, consider `Tooltip` or `Popover` as an alternative for critical information.
- The card is focusable and explorable once open via mouse hover.
