## Usage

```tsx
import { Input } from "@nebutra/custom-ui/primitives"
```

## Examples

### Basic

```tsx
<Input placeholder="Enter value…" />
```

### With Label

```tsx
<div className="flex flex-col gap-1.5">
  <label className="text-sm font-medium" htmlFor="email">
    Email address
  </label>
  <Input id="email" type="email" placeholder="hello@nebutra.com" />
</div>
```

### With Error State

```tsx
<div className="flex flex-col gap-1.5">
  <label className="text-sm font-medium" htmlFor="error-input">
    Email address
  </label>
  <Input
    id="error-input"
    type="email"
    defaultValue="not-an-email"
    className="border-destructive focus-visible:ring-destructive"
    aria-invalid="true"
    aria-describedby="email-error"
  />
  <p id="email-error" className="text-xs text-destructive">
    Please enter a valid email address.
  </p>
</div>
```

### Search with Icon

```tsx
import { Search } from "lucide-react"

<div className="relative">
  <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
  <Input className="pl-9" placeholder="Search…" />
</div>
```

### Controlled

```tsx
const [value, setValue] = useState("")

<Input
  value={value}
  onChange={(e) => setValue(e.target.value)}
  placeholder="Type something…"
/>
```

### Disabled

```tsx
<Input placeholder="Cannot be edited" disabled />
```

## Props

`Input` extends all native `React.InputHTMLAttributes<HTMLInputElement>`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `string` | `"text"` | HTML input type (`text`, `email`, `password`, `number`, `search`, etc.) |
| `placeholder` | `string` | — | Placeholder text |
| `disabled` | `boolean` | `false` | Disables the input |
| `className` | `string` | — | Additional CSS classes |
| `value` | `string \| number` | — | Controlled value |
| `defaultValue` | `string \| number` | — | Uncontrolled default value |
| `onChange` | `ChangeEventHandler<HTMLInputElement>` | — | Change handler |
| `ref` | `Ref<HTMLInputElement>` | — | Forwarded ref |

## Design Tokens

Height locked to `h-10` (40px) from `inputTokens.height`.
Focus ring: `--focus-ring` — brand-blue 2px ring, 2px offset.

## Accessibility

- Renders native `<input>` — always pair with a visible `<label>` or `aria-label`.
- Use `aria-invalid="true"` + `aria-describedby` to associate error messages.
- Disabled state sets `cursor-not-allowed` and reduces opacity.
- Focus ring is always visible for keyboard navigation.
