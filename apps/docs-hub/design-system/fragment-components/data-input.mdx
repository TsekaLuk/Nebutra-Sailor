## Usage

```tsx
import { DataInput } from "@nebutra/custom-ui";

export function IntegrationForm() {
  const [url, setUrl] = useState("");
  const [error, setError] = useState<string | undefined>();

  function handleChange(value: string) {
    setUrl(value);
    setError(value.startsWith("https://") ? undefined : "Must start with https://");
  }

  return (
    <DataInput
      label="Webhook URL"
      placeholder="https://hooks.acme.com/webhook"
      value={url}
      onChange={(e) => handleChange(e.target.value)}
      error={error}
    />
  );
}
```

## Props

| Prop | Type | Default | Description |
|---|---|---|---|
| `label` | `string` | — | Visible label rendered above the input |
| `type` | `React.HTMLInputTypeAttribute` | `"text"` | Native input type |
| `placeholder` | `string` | — | Input placeholder text |
| `value` | `string \| number` | — | Controlled value |
| `defaultValue` | `string \| number` | — | Uncontrolled initial value |
| `onChange` | `React.ChangeEventHandler<HTMLInputElement>` | — | Change handler |
| `prefix` | `string \| React.ReactNode` | — | Content rendered inside the input before the value |
| `suffix` | `string \| React.ReactNode` | — | Content rendered inside the input after the value |
| `helperText` | `string` | — | Muted hint text shown below the input |
| `error` | `string` | — | Validation error message; overrides `helperText` and applies error styles |
| `disabled` | `boolean` | `false` | Disables the input |
| `required` | `boolean` | `false` | Marks the field as required; adds visual indicator |
| `className` | `string` | — | Additional classes on the root wrapper |

## Examples

### Required field with helper text

```tsx
<DataInput
  label="Team name"
  required
  placeholder="e.g. Platform Engineering"
  helperText="This will be visible to all workspace members."
/>
```

### Numeric input with unit suffix

```tsx
<DataInput
  label="Rate limit"
  type="number"
  suffix="req/min"
  defaultValue={100}
  min={1}
  max={10000}
/>
```

### With icon prefix

```tsx
import { Search } from "lucide-react";

<DataInput
  label="Search resources"
  prefix={<Search className="w-4 h-4 text-gray-400" />}
  placeholder="Filter by name..."
/>
```

## Accessibility

- Label is always rendered as a `<label>` element associated to the input via `htmlFor` / `id`.
- Error messages are announced via `aria-describedby` pointing to an element with `role="alert"`.
- Helper text uses `aria-describedby` when no error is present.
- Required fields have `aria-required="true"` on the input; the visual required indicator is `aria-hidden`.
- Prefix and suffix decorators are `aria-hidden="true"` and excluded from tab order.
