---

## Demo

```tsx
import { PageHeader } from "@nebutra/design-system";

<PageHeader
  title="Pipeline settings"
  description="Configure ingestion rules, filters, and destinations."
  breadcrumbs={[
    { label: "Pipelines", href: "/pipelines" },
    { label: "Ingestion pipeline" },
  ]}
  actions={
    <Button variant="primary">Save changes</Button>
  }
/>
```

## Usage

```tsx
import { PageHeader } from "@nebutra/design-system";

export function PipelineSettingsPage({ pipeline }: { pipeline: Pipeline }) {
  return (
    <div className="p-6">
      <PageHeader
        title={pipeline.name}
        description={`Last updated ${formatRelative(pipeline.updatedAt)}`}
        breadcrumbs={[
          { label: "Pipelines", href: "/pipelines" },
          { label: pipeline.name },
        ]}
        actions={
          <div className="flex gap-2">
            <Button variant="secondary" onClick={handleDisable}>Pause</Button>
            <Button onClick={handleSave}>Save</Button>
          </div>
        }
      />

      <PipelineForm pipeline={pipeline} />
    </div>
  );
}
```

## Props

### `PageHeaderProps`

| Prop | Type | Default | Description |
|---|---|---|---|
| `title` | `string` | — | **Required.** Page heading rendered as `<h1>` |
| `description` | `string` | — | Muted subtitle rendered below the title |
| `actions` | `React.ReactNode` | — | Action controls (buttons, menus) anchored to the right on wider screens |
| `breadcrumbs` | `Array<{ label: string; href?: string }>` | — | Breadcrumb trail rendered above the title using `@primer/react` `Breadcrumbs` |
| `className` | `string` | — | Additional classes on the `<header>` root |
| `children` | `React.ReactNode` | — | Additional content rendered below the title row (e.g. tabs) |

### Breadcrumb item

```ts
interface BreadcrumbItem {
  label: string;
  href?: string;   // Omit for the current (last) item
}
```

The last item in the `breadcrumbs` array is automatically marked as `selected` and rendered without a link.

## Examples

### Title only

```tsx
<PageHeader title="Members" />
```

### With tabs below

```tsx
<PageHeader
  title="Analytics"
  description="Usage metrics for the last 30 days."
>
  <Tabs className="mt-4">
    <Tab>Overview</Tab>
    <Tab>Events</Tab>
    <Tab>Errors</Tab>
  </Tabs>
</PageHeader>
```

### Deep breadcrumb trail

```tsx
<PageHeader
  title="Edit alert"
  breadcrumbs={[
    { label: "Dashboard", href: "/" },
    { label: "Monitoring", href: "/monitoring" },
    { label: "Alerts", href: "/monitoring/alerts" },
    { label: "High error rate" },
  ]}
/>
```

### No breadcrumbs, actions only

```tsx
<PageHeader
  title="API keys"
  actions={
    <Button onClick={() => setOpen(true)}>
      Generate key
    </Button>
  }
/>
```

## Accessibility

- The `title` renders as `<h1>` — each page must have exactly one `<h1>`.
- The `<header>` element uses the implicit landmark role `banner`.
- Breadcrumbs render inside `<nav aria-label="Breadcrumb">` (via `@primer/react`).
- The last breadcrumb item has `aria-current="page"`.
- The `actions` slot has no implicit landmark role; wrap with `<nav aria-label="Page actions">` if it contains navigation.
- `description` is plain `<p>` text — do not use it as the sole source of critical instructions.
