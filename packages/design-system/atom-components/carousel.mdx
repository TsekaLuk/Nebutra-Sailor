## Demo

```tsx
import {
  Carousel,
  CarouselContent,
  CarouselItem,
  CarouselNext,
  CarouselPrevious,
} from "@nebutra/ui/primitives"
import { Card, CardContent } from "@nebutra/ui/primitives"

export function CarouselDemo() {
  return (
    <Carousel className="w-full max-w-xs">
      <CarouselContent>
        {Array.from({ length: 5 }).map((_, index) => (
          <CarouselItem key={index}>
            <div className="p-1">
              <Card>
                <CardContent className="flex aspect-square items-center justify-center p-6">
                  <span className="text-4xl font-semibold">{index + 1}</span>
                </CardContent>
              </Card>
            </div>
          </CarouselItem>
        ))}
      </CarouselContent>
      <CarouselPrevious />
      <CarouselNext />
    </Carousel>
  )
}
```

## Usage

```tsx
import {
  Carousel,
  CarouselContent,
  CarouselItem,
  CarouselNext,
  CarouselPrevious,
} from "@nebutra/ui/primitives"
```

## Props

### Carousel (Root)

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Scroll direction |
| `opts` | `EmblaOptionsType` | — | Embla Carousel options |
| `plugins` | `EmblaPluginType[]` | — | Embla Carousel plugins |
| `setApi` | `(api: CarouselApi) => void` | — | Access the Embla carousel API |
| `className` | `string` | — | Additional CSS classes |
| `children` | `ReactNode` | — | Carousel content |

### CarouselItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | — | Additional CSS classes. Use `basis-*` to control item width |
| `children` | `ReactNode` | — | Slide content |

## Examples

### Multiple Items Per View

```tsx
<Carousel opts={{ align: "start" }}>
  <CarouselContent>
    {items.map((item) => (
      <CarouselItem key={item.id} className="basis-1/3">
        <Card>
          <CardContent className="p-4">{item.title}</CardContent>
        </Card>
      </CarouselItem>
    ))}
  </CarouselContent>
  <CarouselPrevious />
  <CarouselNext />
</Carousel>
```

### Vertical Carousel

```tsx
<Carousel orientation="vertical" className="max-h-64">
  <CarouselContent>
    {items.map((item) => (
      <CarouselItem key={item.id}>{item.content}</CarouselItem>
    ))}
  </CarouselContent>
  <CarouselPrevious />
  <CarouselNext />
</Carousel>
```

## Accessibility

- Navigation buttons (`CarouselPrevious`, `CarouselNext`) have descriptive `aria-label` attributes.
- The carousel region should have `aria-label` or `aria-labelledby` when used in context.
- Provide keyboard-accessible alternatives for autoplay carousels.
- Ensure sufficient color contrast on navigation controls.
